import h5py
import xarray as xr
import numpy as np
root = h5py.File('test.h5', 'w')

Create some groups

grp1 = root.create_group("group_1")
grp2 = root.create_group("group_2")

Create some Datasets in those groups

nx1 = 4
ny1 = 3
grp1["data1"] = np.ones((nx1, ny1), 'f')
grp1["data2"] = np.ones((nx1, ny1), 'f')

nx2 = 5
ny2 = 3
grp2["data3"] = np.ones((nx2, ny2), 'i')

Create some dimension variables. The dimension scales need to be in the group so that they can found by xarray.

grp1['x'] = np.arange(nx1)
grp1['x'].make_scale('x')
grp1['y'] = np.arange(ny1)
grp1['y'].make_scale('y')

Attach scales to the group_1 data

grp1["data1"].dims[0].attach_scale(grp1['x'])
grp1["data1"].dims[1].attach_scale(grp1['y'])

grp1["data2"].dims[0].attach_scale(grp1['x'])
grp1["data2"].dims[1].attach_scale(grp1['y'])
list_of_objects = []
root.visit(list_of_objects.append)
list_of_objects
['group_1',
 'group_1/data1',
 'group_1/data2',
 'group_1/x',
 'group_1/y',
 'group_2',
 'group_2/data3']
root.close()

ds = xr.open_dataset("test.h5", group="group_1")
